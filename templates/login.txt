                        <div class="form-actions" style="margin-bottom:20px;">

                        <button type="button" class="btn btn-google" onclick="window.location.href='/google-login'">

                        Already have an account?

                <!-- FORGOT PASSWORD — inside forms-container so grid-area:stack overlay works -->
                <div id="forgotSection" class="form-section hidden slide-right">
                    <form id="forgotForm" novalidate>

                        <div class="form-group">
                            <div class="input-wrapper">
                                <input type="email" id="forgotEmail" class="input-field" placeholder=" " required>
                                <label for="forgotEmail" class="input-label">Enter your registered email</label>
                                <i class="fa-solid fa-envelope input-icon"></i>
                            </div>
                            <span class="error-msg">
                                <i class="fa-solid fa-circle-exclamation" aria-hidden="true"></i>
                                <span class="error-text"></span>
                            </span>
                        </div>

                        <button type="submit" id="forgotBtn" class="btn btn-primary">
                            <i class="fa-solid fa-paper-plane"></i>
                            Send Reset Link
                        </button>
                    </form>

                    <div class="card-footer">
                        Remember your password?
                        <a id="backToLogin" class="link">Sign in</a>
                    </div>
                </div>

            </div><!-- /forms-container -->
            <div class="security-badge">
                <i class="fa-solid fa-shield-halved" aria-hidden="true"></i>
                Security enabled • HTTPS in production
            </div>
        </div><!-- /card -->

    <script>
        'use strict';

        const loginSection = document.getElementById('loginSection');
        const signupSection = document.getElementById('signupSection');
        const showSignUpLink = document.getElementById('showSignUp');
        const showLoginLink = document.getElementById('showLogin');
        const loginForm = document.getElementById('loginForm');
        const signupForm = document.getElementById('signupForm');
        const pageTitle = document.getElementById('pageTitle');
        const pageTagline = document.getElementById('pageTagline');
        const toastContainer = document.getElementById('toastContainer');

        /* ── TOAST ── */
        function showToast(type, title, message, duration = 4200) {
            const icons = { success: 'fa-circle-check', error: 'fa-circle-xmark', warning: 'fa-triangle-exclamation' };
            const t = document.createElement('div');
            t.className = `toast toast-${type}`;
            t.innerHTML = `
            <i class="fa-solid ${icons[type]} toast-icon" aria-hidden="true"></i>
            <div class="toast-content">
                <div class="toast-title">${title}</div>
                <div class="toast-message">${message}</div>
            </div>
            <i class="fa-solid fa-xmark toast-close" role="button" aria-label="Dismiss" tabindex="0"></i>`;
            toastContainer.appendChild(t);
            t.querySelector('.toast-close').addEventListener('click', () => removeToast(t));
            setTimeout(() => removeToast(t), duration);
        }
        function removeToast(t) {
            if (t._r) return; t._r = true;
            t.classList.add('toast-exit');
            setTimeout(() => t.remove(), 300);
        }

        /* ── SWITCH VIEWS ── */
        function switchToSignup() {
            loginSection.classList.add('slide-left', 'hidden');
            forgotSection.classList.add('slide-right', 'hidden');
            signupSection.classList.remove('hidden', 'slide-right');
            pageTitle.textContent = 'Create your account';
            pageTagline.textContent = 'Start your journey to financial freedom';
            setTimeout(() => document.getElementById('fullName').focus(), 340);
        }
        function switchToLogin() {
            signupSection.classList.add('slide-right', 'hidden');
            forgotSection.classList.add('slide-right', 'hidden');
            loginSection.classList.remove('hidden', 'slide-left');
            pageTitle.textContent = 'Welcome back';
            pageTagline.textContent = 'Sign in to continue to your dashboard';
            setTimeout(() => document.getElementById('loginEmail').focus(), 340);
        }

        showSignUpLink.addEventListener('click', e => { e.preventDefault(); switchToSignup(); });
        showSignUpLink.addEventListener('keypress', e => { if (e.key === 'Enter') { e.preventDefault(); switchToSignup(); } });
        showLoginLink.addEventListener('click', e => { e.preventDefault(); switchToLogin(); });
        showLoginLink.addEventListener('keypress', e => { if (e.key === 'Enter') { e.preventDefault(); switchToLogin(); } });

        /* ── PASSWORD TOGGLE ── */
        document.querySelectorAll('.password-toggle').forEach(toggle => {
            const handle = () => {
                const input = document.getElementById(toggle.dataset.target);
                const hide = input.type === 'password';
                input.type = hide ? 'text' : 'password';
                toggle.classList.replace(hide ? 'fa-eye-slash' : 'fa-eye', hide ? 'fa-eye' : 'fa-eye-slash');
            };
            toggle.addEventListener('click', handle);
            toggle.addEventListener('keypress', e => { if (e.key === 'Enter') handle(); });
        });

        /* ── CAPS LOCK ── */
        document.querySelectorAll('input[type="password"]').forEach(input => {
            const warn = input.closest('.form-group').querySelector('.caps-warning');
            if (!warn) return;
            input.addEventListener('keyup', e => {
                warn.classList.toggle('active', !!(e.getModifierState && e.getModifierState('CapsLock')));
            });
        });

        /* ── PASSWORD STRENGTH ── */
        const signupPassInput = document.getElementById('signupPass');
        const strengthBars = document.getElementById('strengthBars');
        const strengthRow = document.getElementById('strengthRow');
        const strengthLabel = document.getElementById('strengthLabel');
        const LABELS = { weak: 'Weak', medium: 'Fair', strong: 'Strong ✓' };

        function calcStrength(pw) {
            let s = 0;
            if (pw.length >= 8) s++;
            if (pw.length >= 12) s++;
            if (/[a-z]/.test(pw) && /[A-Z]/.test(pw)) s++;
            if (/\d/.test(pw)) s++;
            if (/[^a-zA-Z0-9]/.test(pw)) s++;
            return s <= 2 ? 'weak' : s <= 4 ? 'medium' : 'strong';
        }

        signupPassInput.addEventListener('input', () => {
            const pw = signupPassInput.value;
            if (!pw.length) {
                strengthBars.className = 'password-strength';
                strengthRow.classList.remove('active');
                return;
            }
            const level = calcStrength(pw);
            strengthBars.className = `password-strength active ps-${level}`;
            strengthRow.classList.add('active');
            strengthLabel.textContent = LABELS[level];
        });

        /* ── VALIDATION ── */
        function showError(input, msg) {
            const g = input.closest('.form-group');
            g.classList.add('error'); g.classList.remove('success');
            const et = g.querySelector('.error-text');
            if (et && msg) et.textContent = msg;
        }
        function showSuccess(input) {
            const g = input.closest('.form-group');
            g.classList.remove('error'); g.classList.add('success');
        }
        function clearErrors(form) {
            form.querySelectorAll('.form-group').forEach(g => g.classList.remove('error', 'success'));
        }
        function isValidEmail(e) { return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(e); }

        [document.getElementById('loginEmail'), document.getElementById('signupEmail')].forEach(input => {
            input.addEventListener('blur', () => {
                if (input.value && !isValidEmail(input.value)) showError(input, 'Please enter a valid email address');
                else if (input.value) showSuccess(input);
            });
            input.addEventListener('input', () => input.closest('.form-group').classList.remove('error'));
        });
        document.querySelectorAll('.input-field').forEach(input => {
            input.addEventListener('input', () => input.closest('.form-group').classList.remove('error'));
        });

        /* ── LOGIN SUBMIT ── */
        let isLoginSubmitting = false;
        loginForm.addEventListener('submit', async e => {
            e.preventDefault();
            if (isLoginSubmitting) return;
            clearErrors(loginForm);
            const emailInput = document.getElementById('loginEmail');
            const passInput = document.getElementById('loginPass');
            let valid = true;
            if (!isValidEmail(emailInput.value.trim())) { showError(emailInput, 'Please enter a valid email address'); valid = false; }
            if (!passInput.value) { showError(passInput, 'Password is required'); valid = false; }
            if (!valid) return;

            isLoginSubmitting = true;
            const btn = loginForm.querySelector('button[type="submit"]');
            const orig = btn.innerHTML;
            btn.disabled = true; btn.innerHTML = '<span class="spinner"></span> Signing in…';

            try {
                const res = await fetch('/do-login', {
                    method: 'POST', headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ email: emailInput.value.trim(), password: passInput.value })
                });
                const data = await res.json();
                if (res.ok && data.success) {
                    showToast('success', 'Welcome back!', 'Redirecting to your dashboard…');
                    setTimeout(() => { window.location.href = '/'; }, 1100);
                } else {
                    showError(emailInput, ''); showError(passInput, '');
                    showToast('error', 'Login Failed', data.message || 'Invalid email or password');
                }
            } catch (err) {
                showToast('error', 'Network Error', 'Unable to connect. Please check your connection.');
            } finally {
                isLoginSubmitting = false; btn.disabled = false; btn.innerHTML = orig;
            }
        });

        const forgotSection = document.getElementById('forgotSection');
        const forgotLink = document.getElementById('forgotPasswordLink');
        const backToLogin = document.getElementById('backToLogin');

        function switchToForgot() {
            loginSection.classList.add('slide-left', 'hidden');
            signupSection.classList.add('hidden');
            forgotSection.classList.remove('hidden', 'slide-right');
            pageTitle.textContent = 'Reset your password';
            pageTagline.textContent = 'We will send you a reset link';
        }

        forgotLink.addEventListener('click', e => {
            e.preventDefault();
            switchToForgot();
        });

        backToLogin.addEventListener('click', e => {
            e.preventDefault();
            switchToLogin();
        });

        const forgotForm = document.getElementById('forgotForm');

        forgotForm.addEventListener('submit', async e => {
            e.preventDefault();

            const emailInput = document.getElementById('forgotEmail');
            const email = emailInput.value.trim();

            if (!isValidEmail(email)) {
                showError(emailInput, 'Please enter a valid email address');
                return;
            }

            const btn = document.getElementById('forgotBtn');
            const orig = btn.innerHTML;
            btn.disabled = true;
            btn.innerHTML = '<span class="spinner"></span> Sending…';

            try {
                await fetch('/forgot-password', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ email })
                });
            } catch {
                // Swallow network errors — always show the same message (security)
            } finally {
                btn.disabled = false;
                btn.innerHTML = orig;
            }

            // Always show success — never reveal whether the email exists
            showToast('success', 'Reset Link Sent', 'If that email exists, a reset link has been sent.');
            forgotForm.reset();
            // Stay on page — user can go back manually via "Sign in" link
        });

        /* ── SIGNUP SUBMIT ── */
        let isSignupSubmitting = false;
        signupForm.addEventListener('submit', async e => {
            e.preventDefault();
            if (isSignupSubmitting) return;
            clearErrors(signupForm);
            const nameInput = document.getElementById('fullName');
            const emailInput = document.getElementById('signupEmail');
            const passInput = document.getElementById('signupPass');
            const terms = document.getElementById('agreeTerms');
            let valid = true;
            if (!nameInput.value.trim()) { showError(nameInput, 'Full name is required'); valid = false; }
            if (!isValidEmail(emailInput.value.trim())) { showError(emailInput, 'Please enter a valid email address'); valid = false; }
            if (passInput.value.length < 6) { showError(passInput, 'Password must be at least 6 characters'); valid = false; }
            if (!terms.checked) { showToast('warning', 'Terms Required', 'You must agree to the Terms & Privacy policy'); valid = false; }
            if (!valid) return;

            isSignupSubmitting = true;
            const btn = signupForm.querySelector('button[type="submit"]');
            const orig = btn.innerHTML;
            btn.disabled = true; btn.innerHTML = '<span class="spinner"></span> Creating account…';

            try {
                const res = await fetch('/signup', {
                    method: 'POST', headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ name: nameInput.value.trim(), email: emailInput.value.trim(), password: passInput.value })
                });
                const data = await res.json();
                if (res.ok && data.success) {
                    showToast('success', 'Account Created!', 'Welcome to Budgetly. Please sign in to continue.');
                    signupForm.reset();
                    strengthBars.className = 'password-strength';
                    strengthRow.classList.remove('active');
                    setTimeout(() => switchToLogin(), 1100);
                } else {
                    if (data.message && data.message.toLowerCase().includes('email')) showError(emailInput, data.message);
                    else showToast('error', 'Signup Failed', data.message || 'Please try again');
                }
            } catch (err) {
                showToast('error', 'Network Error', 'Unable to connect. Please check your connection.');
            } finally {
                isSignupSubmitting = false; btn.disabled = false; btn.innerHTML = orig;
            }
        });

        document.addEventListener('keydown', e => { if (e.key === 'Escape') document.activeElement.blur(); });
        window.addEventListener('load', () => { setTimeout(() => document.getElementById('loginEmail').focus(), 580); });
    </script>
